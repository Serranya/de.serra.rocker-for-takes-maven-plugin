1   package de.serra.rft;
2   
3   import org.apache.maven.plugin.AbstractMojo;
4   import org.apache.maven.plugin.MojoExecutionException;
5   import org.apache.maven.plugin.MojoFailureException;
6   import org.apache.maven.plugins.annotations.LifecyclePhase;
7   import org.apache.maven.plugins.annotations.Mojo;
8   import org.apache.maven.plugins.annotations.Parameter;
9   import com.fizzed.rocker.compiler.RockerConfiguration;
10  import com.fizzed.rocker.compiler.RockerUtil;
11  import com.fizzed.rocker.compiler.TemplateParser;
12  import com.fizzed.rocker.model.TemplateModel;
13  import com.fizzed.rocker.runtime.ParserException;
14  
15  import de.serra.rft.DefaultResponseClass.Flag;
16  
17  import java.io.File;
18  import java.io.IOException;
19  import java.util.Collection;
20  import java.util.EnumSet;
21  import java.util.Set;
22  import java.util.function.Predicate;
23  import java.util.regex.Pattern;
24  import java.util.stream.Collectors;
25  
26  @Mojo(name = "generate", defaultPhase = LifecyclePhase.GENERATE_SOURCES, threadSafe = true)
27  public class RockerResponseMojo extends AbstractMojo {
28  	/**
29  	 * Whether the plugin should let the build fail if it encounters an error.
30  	 */
31  	@Parameter(property = "rft.failOnError", defaultValue = "true")
32  	protected boolean failOnError;
33  
34  	/**
35  	 * Directory containing templates. The base directory to search -- which is also
36  	 * how their "package" name is determined.
37  	 */
38  	@Parameter(property = "rft.templateDirectory", defaultValue = "${project.build.sourceDirectory}")
39  	protected File templateDirectory;
40  
41  	/**
42  	 * Directory to output generated Java source files.
43  	 */
44  	@Parameter(property = "rft.outputDirectory",
45  			defaultValue = "${project.build.directory}/generated-sources/rocker", required = true)
46  	protected File outputDirectory;
47  
48  	/**
49  	 * The regex to recognize rocker template files.
50  	 */
51  	@Parameter(property = "rft.suffixRegex", defaultValue = ".*\\.rocker\\.(raw|html)$")
52  	protected String suffixRegex;
53  
54  	/**
55  	 * Whether {@code @EqualsAndHashCode} from lombok should be generated.
56  	 */
57  	@Parameter(property = "rft.useLombok", defaultValue = "true")
58  	protected boolean useLombok;
59  
60  	/**
61  	 * Whether {@code @SuppressFBWarnings} annotations should be generated.
62  	 */
63  	@Parameter(property = "rft.useSupressFbWarnings", defaultValue = "true")
64  	protected boolean useSupressFbWarnings;
65  
66  	public void execute() throws MojoExecutionException, MojoFailureException {
67  		if (templateDirectory == null) {
68  			throw new MojoExecutionException("Property templateDirectory cannot be null/empty");
69  		}
70  		if (this.outputDirectory == null) {
71  			throw new MojoExecutionException("Property outputDirectory cannot be null/empty");
72  		}
73  
74  		Collection<File> allFiles = RockerUtil.listFileTree(templateDirectory);
75  		Predicate<String> suffixPredicate =
76  				Pattern.compile(suffixRegex == null ? "rocker.html" : suffixRegex).asMatchPredicate();
77  		Collection<File> templateFiles =
78  				allFiles.stream().filter(f -> suffixPredicate.test(f.getName())).collect(Collectors.toList());
79  
80  		getLog().info("Parsing " + templateFiles.size() + " rocker template files");
81  
82  		int errors = 0;
83  		int generated = 0;
84  		TemplateParser parser = new TemplateParser(new RockerConfiguration());
85  		parser.getConfiguration().setTemplateDirectory(templateDirectory);
86  		parser.getConfiguration().setOutputDirectory(outputDirectory);
87  
88  		for (File f : templateFiles) {
89  			TemplateModel model = null;
90  			try {
91  				// parse model
92  				model = parser.parse(f);
93  			} catch (ParserException e) {
94  				getLog().error("Parsing failed for " + f + ":[" + e.getLineNumber() + "," + e.getColumnNumber()
95  						+ "] " + e.getMessage());
96  				if (failOnError) {
97  					throw new MojoExecutionException("Error while parsing templates", e);
98  				}
99  				errors++;
100 			} catch (IOException e) {
101 				getLog().error("Unable to parse template", e);
102 				if (failOnError) {
103 					throw new MojoExecutionException("Error while parsing templates", e);
104 				}
105 				errors++;
106 			}
107 			try {
108 				Set<Flag> flags = EnumSet.noneOf(Flag.class);
109 				if (useLombok) {
110 					flags.add(Flag.LOMBOK_EQUALS_AND_HASCODE);
111 				}
112 				if (useSupressFbWarnings) {
113 					flags.add(Flag.SUPPRESS_FB_WARNINGS);
114 				}
115 				if (model != null) {
116 					new FileWritingResponseClass(model, outputDirectory.toPath(), flags).generate();
117 					generated++;
118 				}
119 			} catch (IOException e) {
120 				getLog().error("Unable to generate responseClass", e);
121 				if (failOnError) {
122 					throw new MojoExecutionException("Unable to generate responseClass", e);
123 				}
124 				errors++;
125 			}
126 		}
127 
128 		getLog().info("Generated " + generated + " rocker response java source files");
129 
130 		if (errors > 0 && failOnError) {
131 			throw new RuntimeException("Caught " + errors + " errors.");
132 		}
133 	}
134 }
