3   import org.apache.maven.plugin.AbstractMojo;
4   import org.apache.maven.plugin.MojoExecutionException;
5   import org.apache.maven.plugin.MojoFailureException;
6   import org.apache.maven.plugins.annotations.LifecyclePhase;
7   import org.apache.maven.plugins.annotations.Mojo;
8   import org.apache.maven.plugins.annotations.Parameter;
9   import com.fizzed.rocker.compiler.RockerConfiguration;
10  import com.fizzed.rocker.compiler.RockerUtil;
11  import com.fizzed.rocker.compiler.TemplateParser;
12  import com.fizzed.rocker.model.TemplateModel;
13  import com.fizzed.rocker.runtime.ParserException;
14  
15  import de.serra.rft.DefaultResponseClass.Flag;
16  
17  import java.io.File;
18  import java.io.IOException;
19  import java.util.Collection;
20  import java.util.EnumSet;
21  import java.util.List;
22  import java.util.Set;
23  import java.util.function.Predicate;
24  import java.util.regex.Pattern;
25  import java.util.stream.Collectors;
26  
27  @Mojo(name = "generate", defaultPhase = LifecyclePhase.GENERATE_SOURCES, threadSafe = true)
28  public class RockerResponseMojo extends AbstractMojo {
29  	/**
30  	 * Whether the plugin should let the build fail if it encounters an error.
31  	 */
32  	@Parameter(property = "rft.failOnError", defaultValue = "true")
33  	protected boolean failOnError;
34  
35  	/**
36  	 * Directory containing templates. The base directory to search -- which is also
37  	 * how their "package" name is determined.
38  	 */
39  	@Parameter(property = "rft.templateDirectory", defaultValue = "${project.build.sourceDirectory}")
40  	protected File templateDirectory;
41  
42  	/**
43  	 * Directory to output generated Java source files.
44  	 */
45  	@Parameter(property = "rft.outputDirectory",
46  			defaultValue = "${project.build.directory}/generated-sources/rocker", required = true)
47  	protected File outputDirectory;
48  
49  	/**
50  	 * The regex to recognize rocker template files.
51  	 */
52  	@Parameter(property = "rft.suffixRegex", defaultValue = ".*\\.rocker\\.(raw|html)$")
53  	protected String suffixRegex;
54  
55  	/**
56  	 * Whether {@code @EqualsAndHashCode} from lombok should be generated.
57  	 */
58  	@Parameter(property = "rft.useLombok", defaultValue = "true")
59  	protected boolean useLombok;
60  
61  	/**
62  	 * Whether {@code @SuppressFBWarnings} annotations should be generated.
63  	 */
64  	@Parameter(property = "rft.useSupressFbWarnings", defaultValue = "true")
65  	protected boolean useSupressFbWarnings;
66  
67  	/**
68  	 * Adds the specified annotation to all generated classes. Can be used to add a custom {@code @Generated} annotation.
69  	 */
70  	@Parameter(property = "rft.annotations")
71  	protected List<String> annotations;
72  
73  	public void execute() throws MojoExecutionException, MojoFailureException {
74  		if (templateDirectory == null) {
75  			throw new MojoExecutionException("Property templateDirectory cannot be null/empty");
76  		}
77  		if (this.outputDirectory == null) {
78  			throw new MojoExecutionException("Property outputDirectory cannot be null/empty");
79  		}
80  
81  		Collection<File> allFiles = RockerUtil.listFileTree(templateDirectory);
82  		Predicate<String> suffixPredicate =
83  				Pattern.compile(suffixRegex == null ? "rocker.html" : suffixRegex).asMatchPredicate();
84  		Collection<File> templateFiles =
85  				allFiles.stream().filter(f -> suffixPredicate.test(f.getName())).collect(Collectors.toList());
86  
87  		getLog().info("Parsing " + templateFiles.size() + " rocker template files");
88  
89  		int errors = 0;
90  		int generated = 0;
91  		TemplateParser parser = new TemplateParser(new RockerConfiguration());
92  		parser.getConfiguration().setTemplateDirectory(templateDirectory);
93  		parser.getConfiguration().setOutputDirectory(outputDirectory);
94  
95  		for (File f : templateFiles) {
96  			TemplateModel model = null;
97  			try {
98  				// parse model
99  				model = parser.parse(f);
100 			} catch (ParserException e) {
101 				getLog().error("Parsing failed for " + f + ":[" + e.getLineNumber() + "," + e.getColumnNumber()
102 						+ "] " + e.getMessage());
103 				if (failOnError) {
104 					throw new MojoExecutionException("Error while parsing templates", e);
105 				}
106 				errors++;
107 			} catch (IOException e) {
108 				getLog().error("Unable to parse template", e);
109 				if (failOnError) {
110 					throw new MojoExecutionException("Error while parsing templates", e);
111 				}
112 				errors++;
113 			}
114 			try {
115 				Set<Flag> flags = EnumSet.noneOf(Flag.class);
116 				if (useLombok) {
117 					flags.add(Flag.LOMBOK_EQUALS_AND_HASCODE);
118 				}
119 				if (useSupressFbWarnings) {
120 					flags.add(Flag.SUPPRESS_FB_WARNINGS);
121 				}
122 				if (model != null) {
123 					new FileWritingResponseClass(model, outputDirectory.toPath(), flags, annotations).generate();
124 					generated++;
125 				}
126 			} catch (IOException e) {
127 				getLog().error("Unable to generate responseClass", e);
128 				if (failOnError) {
129 					throw new MojoExecutionException("Unable to generate responseClass", e);
130 				}
131 				errors++;
132 			}
133 		}
134 
135 		getLog().info("Generated " + generated + " rocker response java source files");
136 
137 		if (errors > 0 && failOnError) {
138 			throw new RuntimeException("Caught " + errors + " errors.");
139 		}
140 	}
141 }
