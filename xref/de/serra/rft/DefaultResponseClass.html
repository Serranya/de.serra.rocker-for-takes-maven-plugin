7   import java.io.BufferedWriter;
8   import java.io.IOException;
9   import java.io.OutputStream;
10  import java.io.OutputStreamWriter;
11  import java.io.Writer;
12  import java.nio.charset.StandardCharsets;
13  import java.util.Collections;
14  import java.util.Set;
15  
16  @SuppressWarnings("checkstyle:MultipleStringLiterals")
17  public class DefaultResponseClass implements ResponseClass {
18  	private final TemplateModel model;
19  	private final OutputStream out;
20  	private final Set<Flag> flags;
21  
22  	public enum Flag {
23  		SUPPRESS_FB_WARNINGS, LOMBOK_EQUALS_AND_HASCODE
24  	}
25  
26  	public DefaultResponseClass(TemplateModel model, OutputStream out) {
27  		this(model, out, Collections.emptySet());
28  	}
29  
30  	public DefaultResponseClass(TemplateModel model, OutputStream out, Set<Flag> flags) {
31  		this.model = model;
32  		this.out = out;
33  		this.flags = flags;
34  	}
35  
36  	@Override
37  	public void generate() throws IOException {
38  		if (model.hasRockerBodyArgument()) {
39  			return;
40  		}
41  
42  		try (Writer w = new BufferedWriter(new OutputStreamWriter(out, StandardCharsets.UTF_8))) {
43  			createSourceTemplate(w);
44  			w.flush();
45  		}
46  	}
47  
48  	private void createSourceTemplate(Writer w) throws IOException {
49  		if (model.getPackageName() != null && !"".equals(model.getPackageName())) {
50  			w.append("package ").append(model.getPackageName()).append(";").append("\n");
51  		}
52  
53  		w.append("\n");
54  		w.append("import com.fizzed.rocker.runtime.ArrayOfByteArraysOutput;\n");
55  		if (flags.contains(Flag.SUPPRESS_FB_WARNINGS)) {
56  			w.append("import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;\n");
57  		}
58  		if (flags.contains(Flag.LOMBOK_EQUALS_AND_HASCODE)) {
59  			w.append("import lombok.EqualsAndHashCode;\n");
60  		}
61  		w.append("import org.takes.Response;\n");
62  		w.append("import org.takes.rs.RsEmpty;\n");
63  		w.append("import org.takes.rs.RsWrap;\n");
64  		w.append("import java.io.ByteArrayInputStream;");
65  		w.append("import java.io.IOException;\n");
66  		w.append("import java.io.InputStream;\n");
67  
68  		w.append("\n");
69  
70  		for (JavaImport i : model.getImports()) {
71  			w.append("import ").append(i.getStatement()).append(";\n");
72  		}
73  
74  		w.append("\n");
75  
76  		if (flags.contains(Flag.SUPPRESS_FB_WARNINGS)) {
77  			w.append("@SuppressFBWarnings(value = \"USBR_UNNECESSARY_STORE_BEFORE_RETURN\", "
78  					+ "justification = \"generated code\")\n");
79  		}
80  		if (flags.contains(Flag.LOMBOK_EQUALS_AND_HASCODE)) {
81  			w.append("@EqualsAndHashCode(callSuper = true)\n");
82  		}
83  		w.append("public class Rs").append(model.getName()).append(" extends RsWrap {\n");
84  		for (Argument arg : model.getArgumentsWithoutRockerBody()) {
85  			w.append("\tprivate final ").append(arg.getType()).append(" ").append(arg.getName()).append(";\n");
86  		}
87  
88  		w.append("\n");
89  
90  		w.append("\tpublic Rs").append(model.getName()).append("(");
91  		boolean firstCArg = true;
92  		for (Argument arg : model.getArgumentsWithoutRockerBody()) {
93  			if (!firstCArg) {
94  				w.append(", ");
95  			} else {
96  				firstCArg = false;
97  			}
98  			w.append("final ").append(arg.getType()).append(" ").append(arg.getName());
99  		}
100 		w.append(") {\n");
101 		w.append("\t\tsuper(new Response() {\n");
102 		w.append("\n");
103 		w.append("\t\t\t@Override\n");
104 		w.append("\t\t\tpublic Iterable<String> head() {\n");
105 		w.append("\t\t\t\treturn new RsEmpty().head();\n");
106 		w.append("\t\t\t}\n");
107 		w.append("\n");
108 		w.append("\t\t\t@Override\n");
109 		w.append("\t\t\t public InputStream body() throws IOException {\n");
110 		w.append("\t\t\t\treturn new ").append("ByteArrayInputStream(\n");
111 		w.append("\t\t\t\t\tnew ").append(model.getName()).append("()\n");
112 		for (Argument arg : model.getArgumentsWithoutRockerBody()) {
113 			w.append("\t\t\t\t\t\t\t.").append(arg.getName()).append("(").append(arg.getName()).append(")\n");
114 		}
115 		w.append("\t\t\t\t\t\t\t.render(ArrayOfByteArraysOutput.FACTORY).toByteArray());\n");
116 		w.append("\t\t\t}\n");
117 		w.append("\t\t});\n");
118 		for (Argument arg : model.getArgumentsWithoutRockerBody()) {
119 			w.append("\t\tthis.").append(arg.getName()).append(" = ").append(arg.getName()).append(";\n");
120 		}
121 		w.append("\t}\n");
122 
123 		for (Argument arg : model.getArgumentsWithoutRockerBody()) {
124 			w.append("\n");
125 			w.append("\tpublic ").append(arg.getType()).append(" ").append(arg.getName()).append("() {\n");
126 			w.append("\t\treturn ").append(arg.getName()).append(";\n");
127 			w.append("\t}\n");
128 		}
129 
130 		w.append("}\n");
131 	}
132 }
